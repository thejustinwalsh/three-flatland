This guide shows you how to create your first sprite using three-flatland.

## Create a Sprite

<Tabs syncKey="framework">
  <TabItem label="Three.js" icon="seti:javascript">
    ```typescript
    import * as THREE from 'three/webgpu';
    import { Sprite2D, Renderer2D } from '@three-flatland/core';

    // Create WebGPU renderer
    const renderer = new THREE.WebGPURenderer();
    await renderer.init();
    document.body.appendChild(renderer.domElement);

    // Create scene and camera
    const scene = new THREE.Scene();
    const camera = new THREE.OrthographicCamera(-1, 1, 1, -1, 0.1, 10);
    camera.position.z = 1;

    // Load texture
    const textureLoader = new THREE.TextureLoader();
    const texture = await textureLoader.loadAsync('/sprite.png');

    // Create 2D renderer and sprite
    const renderer2D = new Renderer2D();
    scene.add(renderer2D);

    const sprite = new Sprite2D({ texture });
    renderer2D.add(sprite);

    // Animation loop
    function animate() {
      requestAnimationFrame(animate);
      renderer.render(scene, camera);
    }
    animate();
    ```
  </TabItem>
  <TabItem label="React" icon="seti:react">
    ```tsx
    import { Canvas, extend } from '@react-three/fiber/webgpu';
    import { Sprite2D, Renderer2D } from '@three-flatland/react';
    import { useTexture } from '@react-three/drei';

    extend({ Sprite2D, Renderer2D });

    function Sprite() {
      const texture = useTexture('/sprite.png');

      return (
        <renderer2D>
          <sprite2D texture={texture} />
        </renderer2D>
      );
    }

    export default function App() {
      return (
        <Canvas orthographic camera={{ zoom: 100 }}>
          <Sprite />
        </Canvas>
      );
    }
    ```
  </TabItem>
